Основной синтаксис:

название_звука
{
	volume=0.5;//Громкость звука 0..1 (По умолчанию: 1)
	volmin=0.1;//Минимальная громкость звука 0..1 (По умолчанию: 0)
	sound="dir\\a.wav";//Название звука (таких полей может быть несколько, тогда поле выбирается по rnd)
	radius=0..+oo;//Величина объекта (только в 3D) с этого момента громкость звука начинает убывать (По умолчанию: 50)


	//Максимальное количество одновременно звучащих звуков (По умолчанию: 5)
	//В случае 2D при попытке загрузить больше, чем max_num_sound звуков - звук не запускается.
	//В случае 3D:
	//    если max_num_sound==1, запускается только 1 звук, последующие не запускаются.
	//    если max_num_sound>1, запускается все звуки, но слышны только ближайшие.

	max_num_sound=0..10;
	max_radius=xxx;//Максимальное расстояние до объекта, после которого звук перестаёт быть слышимым (По умолчанию: oo)
//	starttime=0..1000;//Задержка (в милисекундах) перед началом проигрывания звука (По умолчанию: 0)

//	priority=0..+oo;//Чем больше это число, тем больше вероятность того, 
				//что этот звук будет проигрываться
				//1 - значение по умолчанию, разумный диапазон изменений 
				//(в 3D) - 0.3..3.0
				//(в 2D) - -100..+100

	delta {
		random;//присутствует-выбирать частоту звука случайно при его запуске
			//отсутствует - диапазон изменения частоты в программе
				
		up=0..2;//Во сколько раз частота звука может увеличиваться (По умолчанию: 1.5)
		down=0..20;//Во сколько раз частота звука может увеличиться (По умолчанию: 0.5)

		//speed - больше нельзя использовать. 
		speed=0.01..100;//Скорость, с которой изменяется частота (оборотов в секунду)
	};
};
